					# -o / --output : latex output file
					# -t / --type : az meresi tipus {pkg,ram,core,gpu,all}, egyszerre csak egy, all=pkg + ram
					# -l / --logscale: az abrazolt grafikon y tengelye logaritmikus lesz
					# -c / --correlation: a korreláció és egyéb számítások logolásához a fájlnév pl. ./log.txt
